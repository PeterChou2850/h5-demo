<p>Passage: There are many types of renewable energy...</p >
<p>Question: Which of the following is NOT a renewable energy source?</p >
<form id="quizForm">
    <input type="radio" name="q1" value="A"> Solar Energy<br>
    <input type="radio" name="q1" value="B"> Wind Energy<br>
    <input type="radio" name="q1" value="C"> Coal<br>
    <input type="radio" name="q1" value="D"> Hydro Energy<br>
    <button type="button" onclick="checkAnswer()">提交</button>
</form>
<p id="result"></p >

<script>
    function checkAnswer(){
      var form = document.getElementById('quizForm');
      var answer = form.q1.value;
      if(answer === "C"){
        document.getElementById('result').innerText = "✅ 正确！煤不是可再生能源";
      } else {
        document.getElementById('result').innerText = "❌ 错误，正确答案是C";
      }
    }
</script>
<h3>作业打卡</h3>
<input type="text" id="name" placeholder="输入名字">
<textarea id="work" placeholder="写下今天完成的作业"></textarea><br>
<button onclick="submitWork()">提交</button>
<p id="message"></p >

<script>
    function submitWork(){
      var name = document.getElementById('name').value.trim();
      var work = document.getElementById('work').value.trim();
      if(!name || !work){
        alert("请填写姓名和作业内容");
        return;
      }
      var now = new Date();
      document.getElementById('message').innerText = `✅ ${name} 打卡成功！时间：${now.toLocaleTimeString()}`;
      console.log(name, work, now); // 这里可以上传到云端
      document.getElementById('name').value = "";
      document.getElementById('work').value = "";
    }
</script>
